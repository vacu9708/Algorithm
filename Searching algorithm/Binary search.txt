#include <iostream>
#include <vector>
#include <string>
using namespace std;

string binary_search(vector<int>& list, int target) {
	int first = 0, last = list.size() - 1;
	while (first <= last) {
		int mid = (first + last) / 2;
		if (target == list[mid])
			return "At index " + to_string(mid) + " found";
		if (target < list[mid])
			last = mid - 1;
		if (target > list[mid])
			first = mid + 1;
	}
	return "Not found";
}

void insert_in_order(vector<int>& list, int data) {
	if (list.empty())
		list.push_back(data);
	else {
		int i = 0;
		for (i = 0; i < list.size(); i++)
			if (data < list[i])
				break;
		list.insert(list.begin() + i, data);
	}
}

void print_list(vector<int>& list) {
	for (auto i : list)
		cout << i << " ";
	cout << endl;
}

int main() {
	vector<int> arr_list = { 1,3,5,7 };
	cout << binary_search(arr_list, 2) << endl;
	insert_in_order(arr_list, 2);
	print_list(arr_list);
	cout << binary_search(arr_list, 2) << endl;
	arr_list.erase(arr_list.begin() + 2);
	print_list(arr_list);
}
