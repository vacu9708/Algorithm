배열을 이용하기 때문에 정수만 정렬할 수 있다. 최댓값이 크면 비효율적이다.

----- Code
#include <iostream>
using namespace std;
void counting_sort(int* arr, int length) {
    // Set min and max of the arr
    int max = arr[0];
    int min = arr[0];
    for (int i = 1; i < length; i++)
        if (arr[i] > max)
            max = arr[i];
    int* count = new int[max + 1]{};

    for (int i = 0; i < length; i++) // Sort
        count[arr[i]]++;

    for (int i = 0; i <= max; i++) // Print the result
        for (int j = 0; j < count[i]; j++)
            cout << i << " ";
    delete[] count;
}

int main() {
    int arr[8] = { 5,4,3,2,5,4,3,2 };
    int length = sizeof(arr) / sizeof(arr[0]);
    counting_sort(arr, length);
}
