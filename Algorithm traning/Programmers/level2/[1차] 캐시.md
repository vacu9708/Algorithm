## Using list
~~~python
def solution(cacheSize, cities):
    # edge case
    if cacheSize==0: return 5*len(cities)

    answer = 0
    order = [] # maintains the usage order
    for city in cities:
        city = city.lower()
        try: i=order.index(city)
        except: i=-1
        # Cache miss
        if i==-1:
            answer += 5
            # Remove the oldest if cache is full
            if len(order) == cacheSize:
                order.pop(0)
        # Cache hit
        else:
            answer += 1
            order.pop(i)
        order.append(city) # Latest
    return answer
~~~
