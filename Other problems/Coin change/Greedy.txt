Greedy solution : Fast but doesn't always make the best solution

-----C++
#include <iostream>

using namespace std;

void coin_change(int price, int payed_money) {
	int thousand_won = 0, five_hundred_won = 0, one_hundred_won = 0;
	int change = price - payed_money;

	thousand_won = change / 1000;
	change -= thousand_won * 1000;

	five_hundred_won = change / 500;
	change -= five_hundred_won * 500;

	one_hundred_won = change / 100;
	change -= one_hundred_won * 100;

	cout << thousand_won << " " << five_hundred_won << " " << one_hundred_won;
}

int main() {
	coin_change(10000, 7500);
}
