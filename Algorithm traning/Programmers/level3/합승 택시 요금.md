## Dijikstra
~~~python
class Path:
    def __init__(self,dest,weight):
        self.dest=dest
        self.weight=weight

import heapq
def solution(n, s, a, b, fares):
    # Adjacency list
    adjacencies=[[] for i in range(n+1)]
    for fare in fares:
        adjacencies[fare[0]].append(Path(fare[1],fare[2]))
        adjacencies[fare[1]].append(Path(fare[0],fare[2]))
    # Dijikstra
    INF=999999999
    weight_table=[[INF for j in range(n+1)] for i in range(n+1)]
    for start in range(1,n+1):
        pq=[(0,start)] # weight, destination
        weight_table[start][start]=0
        while pq:
            curr=heapq.heappop(pq) # weight, destination
            for path in adjacencies[curr[1]]:
                new_path_weight=curr[0]+path.weight
                if new_path_weight<weight_table[start][path.dest]:
                    weight_table[start][path.dest]=new_path_weight
                    heapq.heappush(pq,(new_path_weight,path.dest))
    # Answer
    min_cost=99999999999
    for stop in range(1,n+1):
        min_cost=min(min_cost, weight_table[s][stop]+weight_table[stop][a]+weight_table[stop][b])
    return min_cost
~~~
