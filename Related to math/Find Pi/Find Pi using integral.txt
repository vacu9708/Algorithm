Principle : The area of a circle whose radius is 1 is equal to pi.

-----C++
#include <iostream>
#include <random>
#include <time.h>

using namespace std;

void pi_using_integral() {
	double infinity = 654321, dx = 1 / infinity, y = 0, sum = 0;
	for (int i = 0; i < infinity; i++) {
		y = sqrt(1 - pow((i * dx), 2));
		sum += dx * y;
	}

	printf("%.20lf\n", sum * 4);
}

int main() {
	double start = clock();
	pi_using_integral();
	printf("Elapsed time : %lf\n", clock() - start);
}
