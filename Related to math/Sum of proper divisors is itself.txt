#include <iostream>
#include <random>
using namespace std;

void the_sum_of_divisors_of_a_number_is_itself(int range) {
	int count = 0;
	for (int i = 4; i <= range; i += 1) {
		int sum_of_divisors = 0;
		for (int j = 1; j <= i / 2; j += 1) {
			if (i % j == 0) // If the remainder is 0, j is a divisor of i.
				sum_of_divisors += j;
		}
		if (sum_of_divisors == i) { // If the sum of the divisors is itself
			cout << i << " ";
			count += 1;
		}
	}
	cout << "\nTotal number : " << count; // The total number of numbers that have the sum of their proper divisors as itself
}

int main() {
	the_sum_of_divisors_of_a_number_is_itself(1000);
}
