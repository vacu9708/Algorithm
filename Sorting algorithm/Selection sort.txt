#include <iostream>
using namespace std;

void print_array(int* arr, int length) {
	for (int i = 0; i < length; i++)
		cout << arr[i] << " ";
	cout << "\n";
}

void selection_sort(int* arr, int length) {
	for (int i = 0; i < length - 1; i++) {
		int min = i;
		for (int j = i + 1; j < length; j++) // Look for the minimum
			if (arr[j] < arr[min])
				min = j; // Change the current minimum

		int temp = arr[i]; // Swap
		arr[i] = arr[min];
		arr[min] = temp;
		print_array(arr, length); // Show the process
	}
}

void main() {
	int arr[] = { 5,4,3,2,1 };
	int length = sizeof(arr) / sizeof(int);
	selection_sort(arr, length);
}
